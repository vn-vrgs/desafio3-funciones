Requerimiento 1

Función de Expresión

let a = 2;
let b = 3;
let c = 4;
// Función de expresión
function sumar(a, b, c) {
return a + b + c;
};
// Invocando a la función
// guardando el resultado en una variable
let retorno = sumar(a, b, c);
// Enviando por consola varios resultados
console.log("El resultado es: " + retorno );
// Enviar por navegador
alert("El resultado es: " + retorno );
